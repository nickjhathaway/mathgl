MGL code 

rotate 70 70
light on
light 1 1 0 2 'w' 0.3
light 0 0 0 1 'w' 0.3
define x -0.8
define y -0.9
define z 0.2

# fir
cone x y z-1 	x y z 	0.5 0 'g'
cone x y z-0.5 	x y z+0.3	0.4 0 'l'
cone x y z 	x y z+0.6	0.3 0 'g'
cone x y z+0.4 	x y z+0.8 0.2 0 'l'

# decorations
palette 'rbmcnqpy'
for $1 0 8
defpal $2 $1
sphere x+0.4*cos(pi*$1/4) y-0.4*sin(pi*$1/4) z-0.55 0.07 '$2' 
next
for $1 0 6
defpal $2 $1
sphere x+0.3*cos(pi*$1/3+1.4) y-0.3*sin(pi*$1/3+1.4) z-0.05 0.07 '$2' 
next
for $1 0 4
defpal $2 $1
sphere x+0.2*cos(pi*$1/2-2.4) y-0.2*sin(pi*$1/2-2.4) z+0.35 0.07 '$2' 
next
drop x y z+1.1 0 0 -1 0.15 'y'

ambient 0.7
# "snow-woman"
define x -1
define y 0.8
define z 0.4
sphere x y z-1 0.4 'w'
sphere x y z-0.4 0.3 'w'
sphere x y z 0.2 'w'
cone x y z x+0.3 y-0.3 z 0.07 0 'r'
#plot [x+0.2,x] [y,y+0.2] [0.1,0.1] 'k#o'
plot [x+0.18,x+0.08] [y-0.08,y-0.18] [z+0.1,z+0.1] 'k#o '

# ice and snow
new a 50 40
modify a 'x^2/2+y^2/2+0.6*sin(2*pi*x)*sin(3*pi*y) + 0.4*cos(3*pi*(x*y))'
cut off:light off
crange -1 -0.9
surf a*0.1-1 'cwww'

#text
define d 0.02
text -1 1 1 'Happy new year!' 'b:r' -2
text -1 1+d 1-1.3*d 'Happy new year!' 'b:W' -2
